#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_system.h"
#include "esp_wifi.h"
#include "esp_event.h"
#include "esp_log.h"
#include "nvs_flash.h"
#include "esp_http_client.h"
#include "cJSON.h"

static const char *TAG = "HA_CLIENT";

// WiFi settings
#define WIFI_SSID      "Ahmed's S21"
#define WIFI_PASS      "11223344"

// Home Assistant settings
#define HA_IP          "10.222.49.130"
#define HA_PORT        8123
#define HA_ENTRY_ID    "01KBZEDPBKZJ4AS9SGR7MZV4F2"
#define HA_API_KEY     "dev123"

// WiFi event handler
static void wifi_event_handler(void* arg, esp_event_base_t event_base,
                               int32_t event_id, void* event_data) {
    if (event_id == WIFI_EVENT_STA_START) {
        esp_wifi_connect();
    } else if (event_id == WIFI_EVENT_STA_DISCONNECTED) {
        ESP_LOGI(TAG, "WiFi disconnected, reconnecting...");
        vTaskDelay(1000 / portTICK_PERIOD_MS);
        esp_wifi_connect();
    }
}

// IP event handler
static void ip_event_handler(void* arg, esp_event_base_t event_base,
                             int32_t event_id, void* event_data) {
    if (event_id == IP_EVENT_STA_GOT_IP) {
        ip_event_got_ip_t* event = (ip_event_got_ip_t*) event_data;
        ESP_LOGI(TAG, "Got IP: " IPSTR, IP2STR(&event->ip_info.ip));
    }
}

// Initialize WiFi
void wifi_init(void) {
    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    esp_netif_create_default_wifi_sta();

    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&cfg));

    ESP_ERROR_CHECK(esp_event_handler_register(WIFI_EVENT, ESP_EVENT_ANY_ID, &wifi_event_handler, NULL));
    ESP_ERROR_CHECK(esp_event_handler_register(IP_EVENT, IP_EVENT_STA_GOT_IP, &ip_event_handler, NULL));

    wifi_config_t wifi_config = {
        .sta = {
            .ssid = WIFI_SSID,
            .password = WIFI_PASS,
        },
    };

    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config));
    ESP_ERROR_CHECK(esp_wifi_start());

    ESP_LOGI(TAG, "Connecting to WiFi: %s", WIFI_SSID);
}

// HTTP event handler
esp_err_t http_event_handler(esp_http_client_event_t *evt) {
    switch(evt->event_id) {
        case HTTP_EVENT_ON_DATA:
            if (evt->data_len > 0) {
                // Print raw response
                printf("Response: %.*s\n", evt->data_len, (char*)evt->data);
                
                // Parse JSON
                char *json_str = malloc(evt->data_len + 1);
                memcpy(json_str, evt->data, evt->data_len);
                json_str[evt->data_len] = '\0';
                
                cJSON *root = cJSON_Parse(json_str);
                if (root) {
                    cJSON *commands = cJSON_GetObjectItem(root, "commands");
                    if (cJSON_IsArray(commands)) {
                        printf("\n=== COMMANDS ===\n");
                        int size = cJSON_GetArraySize(commands);
                        for (int i = 0; i < size; i++) {
                            cJSON *cmd = cJSON_GetArrayItem(commands, i);
                            printf("Command %d: %s\n", i+1, cmd->valuestring);
                            
                            // Process commands
                            if (strstr(cmd->valuestring, "SET_MODE_")) {
                                printf("  -> Change HVAC mode\n");
                            } else if (strstr(cmd->valuestring, "SET_TEMP_")) {
                                printf("  -> Change temperature\n");
                            } else if (strstr(cmd->valuestring, "SET_FAN_")) {
                                printf("  -> Change fan speed\n");
                            } else if (strcmp(cmd->valuestring, "SWITCH_ON") == 0) {
                                printf("  -> Turn ON switch\n");
                            } else if (strcmp(cmd->valuestring, "SWITCH_OFF") == 0) {
                                printf("  -> Turn OFF switch\n");
                            }
                        }
                        printf("=== END ===\n\n");
                    }
                    cJSON_Delete(root);
                }
                free(json_str);
            }
            break;
        default:
            break;
    }
    return ESP_OK;
}

// Fetch commands from Home Assistant
void fetch_commands(void) {
    char url[256];
    snprintf(url, sizeof(url), 
             "http://%s:%d/api/my_inverter/%s/commands?api_key=%s",
             HA_IP, HA_PORT, HA_ENTRY_ID, HA_API_KEY);
    
    ESP_LOGI(TAG, "Fetching: %s", url);
    
    esp_http_client_config_t config = {
        .url = url,
        .method = HTTP_METHOD_GET,
        .event_handler = http_event_handler,
    };
    
    esp_http_client_handle_t client = esp_http_client_init(&config);
    esp_http_client_set_header(client, "X-API-Key", HA_API_KEY);
    
    esp_err_t err = esp_http_client_perform(client);
    if (err == ESP_OK) {
        ESP_LOGI(TAG, "HTTP GET successful");
    } else {
        ESP_LOGE(TAG, "HTTP GET failed");
    }
    
    esp_http_client_cleanup(client);
}

// Main task
void main_task(void *pvParameter) {
    while(1) {
        fetch_commands();
        vTaskDelay(5000 / portTICK_PERIOD_MS); // 5 seconds
    }
}

void app_main(void) {
    // Initialize NVS
    esp_err_t ret = nvs_flash_init();
    if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
        ESP_ERROR_CHECK(nvs_flash_erase());
        ret = nvs_flash_init();
    }
    ESP_ERROR_CHECK(ret);
    
    // Initialize WiFi
    wifi_init();
    
    // Wait for WiFi connection
    ESP_LOGI(TAG, "Waiting for WiFi...");
    vTaskDelay(5000 / portTICK_PERIOD_MS);
    
    // Start main task
    xTaskCreate(&main_task, "main_task", 4096, NULL, 5, NULL);
}
