/* NEC timing specifications (in microseconds) */
#define NEC_LEADING_CODE_DURATION_0      2500
#define NEC_LEADING_CODE_DURATION_1      1000
#define NEC_PAYLOAD_ZERO_DURATION_0       300
#define NEC_PAYLOAD_ZERO_DURATION_1       400
#define NEC_PAYLOAD_ONE_DURATION_0        300
#define NEC_PAYLOAD_ONE_DURATION_1        900
#define NEC_CONNECTING_CODE1_DURATION_0   300

#define NEC_CONNECTING_CODE2_DURATION_0   2002
#define NEC_CONNECTING_CODE2_DURATION_1   0

#define CMD_QUEUE_LEN 10

/* Global tx_data - protected by mutex */
static char tx_data[65] = {0};
static SemaphoreHandle_t tx_data_mutex = NULL;

/* Command types */
typedef enum {
    CMD_SET_TEMP,
    CMD_SET_MODE,
    CMD_SET_FAN,
    CMD_SET_POWER,
} cmd_type_t;

typedef struct {
    cmd_type_t type;
    int int_value;        // used for temp, fan, power (1/0)
    char mode[16];        // used for mode strings
} ha_cmd_t;

/* IR RX data structure */
typedef struct {
    rmt_symbol_word_t *symbols;
    size_t num_symbols;
} ir_rx_data_t;

/* ---- Utility: bit reverse 4 bits ---- */
static uint8_t bit_reverse4(uint8_t x) {
    return ((x & 1) << 3) | ((x & 2) << 1) | ((x & 4) >> 1) | ((x & 8) >> 3);
}

/* Get nibble (4 bits) from tx_data (encoded as ASCII '0'/'1') */
static uint8_t get_raw(int nib_idx) {
    uint8_t val = 0;
    int s = nib_idx * 4;
    val = (tx_data[s] - '0') << 3 | (tx_data[s + 1] - '0') << 2 |
          (tx_data[s + 2] - '0') << 1 | (tx_data[s + 3] - '0');
    return val;
}

/* Set nibble with raw 4-bit value (thread-safe) */
static void set_nibble(int nib_idx, uint8_t value) {
    if (xSemaphoreTake(tx_data_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
        int s = nib_idx * 4;
        for (int i = 0; i < 4; i++) {
            tx_data[s + i] = ((value >> (3 - i)) & 1) + '0';
        }
        xSemaphoreGive(tx_data_mutex);
    }
}

/* Update checksum nibble (nibble index 15) */
static void update_checksum() {
    if (xSemaphoreTake(tx_data_mutex, pdMS_TO_TICKS(100)) != pdTRUE) {
        return;
    }
    
    int sum = 0;
    for (int n = 0; n < 15; n++) {
        uint8_t raw = get_raw(n);
        uint8_t log_val = bit_reverse4(raw);
        sum += log_val;
    }
    uint8_t log_cs = sum % 16;
    uint8_t raw_cs = bit_reverse4(log_cs);
    
    int s = 15 * 4;
    for (int i = 0; i < 4; i++) {
        tx_data[s + i] = ((raw_cs >> (3 - i)) & 1) + '0';
    }
    
    xSemaphoreGive(tx_data_mutex);
}

/* Set temperature (16..30 permitted in reference) */
void set_temp(int temp) {
    if (temp < 16 || temp > 30) {
        ESP_LOGE(TAG, "Invalid temperature %d", temp);
        return;
    }
    uint8_t log_low = temp % 10;
    uint8_t log_high = temp / 10;
    uint8_t raw_low = bit_reverse4(log_low);
    uint8_t raw_high = bit_reverse4(log_high);
    set_nibble(0, raw_low);
    set_nibble(1, raw_high);
    update_checksum();
    ESP_LOGI(TAG, "set_temp: %d", temp);
}

/* Set mode (off/cool/heat/auto/dry/fan) */
void set_mode(const char* mode) {
    if (strcmp(mode, "off") == 0) {
        // For OFF mode, set power off (bit 3 of nibble 7 = 1)
        uint8_t current_raw7 = get_raw(7);
        uint8_t new_raw7 = current_raw7 | 0b1000;  // set MSB for power off
        set_nibble(7, new_raw7);
        ESP_LOGI(TAG, "set_mode: off (power off)");
    } else if (strcmp(mode, "cool") == 0) {
        // First ensure power is on (clear bit 3 of nibble 7)
        uint8_t current_raw7 = get_raw(7);
        uint8_t new_raw7 = current_raw7 & 0b0111;  // clear MSB for power on
        set_nibble(7, new_raw7);
        set_nibble(6, 0b0100);
    } else if (strcmp(mode, "heat") == 0) {
        uint8_t current_raw7 = get_raw(7);
        uint8_t new_raw7 = current_raw7 & 0b0111;
        set_nibble(7, new_raw7);
        set_nibble(6, 0b0010);
    } else if (strcmp(mode, "auto") == 0) {
        uint8_t current_raw7 = get_raw(7);
        uint8_t new_raw7 = current_raw7 & 0b0111;
        set_nibble(7, new_raw7);
        set_nibble(6, 0b0001);
    } else if (strcmp(mode, "dry") == 0) {
        uint8_t current_raw7 = get_raw(7);
        uint8_t new_raw7 = current_raw7 & 0b0111;
        set_nibble(7, new_raw7);
        set_nibble(6, 0b0000);
        set_nibble(7, 0b0100);
    } else if (strcmp(mode, "fan") == 0) {
        uint8_t current_raw7 = get_raw(7);
        uint8_t new_raw7 = current_raw7 & 0b0111;
        set_nibble(7, new_raw7);
        set_nibble(6, 0b1000);
    } else {
        ESP_LOGE(TAG, "Invalid mode '%s'", mode);
        return;
    }
    update_checksum();
    ESP_LOGI(TAG, "set_mode: %s", mode);
}

/* Set power bit (1=on, 0=off) - uses nibble 7's bits */
void set_power(int on) {
    uint8_t current_raw7 = get_raw(7);
    uint8_t new_raw7;
    if (on) {
        new_raw7 = (current_raw7 & 0b0111); // clear MSB -> power on
    } else {
        new_raw7 = (current_raw7 | 0b1000); // set MSB -> power off
    }
    set_nibble(7, new_raw7);
    update_checksum();
    ESP_LOGI(TAG, "set_power: %d", on);
}

/* Set fan (0: auto, 1: low, 2: med, 3: high) -> nibble 8 */
void set_fan(int level) {
    uint8_t raw;
    switch (level) {
        case 0: raw = 0b1000; break; // auto
        case 1: raw = 0b0001; break; // low
        case 2: raw = 0b0010; break; // med
        case 3: raw = 0b0100; break; // high
        default:
            ESP_LOGE(TAG, "Invalid fan level %d", level);
            return;
    }
    set_nibble(8, raw);
    update_checksum();
    ESP_LOGI(TAG, "set_fan: %d", level);
}

/* Build rmt_symbol_word_t array (waveform) from tx_data string */
static int build_items(rmt_symbol_word_t *items, size_t max_items) {
    if (xSemaphoreTake(tx_data_mutex, pdMS_TO_TICKS(100)) != pdTRUE) {
        return 0;
    }
    
    int idx = 0;
    // Leading code
    items[idx++] = (rmt_symbol_word_t) {
        .level0 = 1, .duration0 = NEC_LEADING_CODE_DURATION_0,
        .level1 = 0, .duration1 = NEC_LEADING_CODE_DURATION_1
    };
    // Payload bits (tx_data is '0'/'1' characters)
    for (int i = 0; tx_data[i] != '\0' && idx < (int)max_items - 4; i++) {
        if (tx_data[i] == '0') {
            items[idx++] = (rmt_symbol_word_t) {
                .level0 = 1, .duration0 = NEC_PAYLOAD_ZERO_DURATION_0,
                .level1 = 0, .duration1 = NEC_PAYLOAD_ZERO_DURATION_1
            };
        } else if (tx_data[i] == '1') {
            items[idx++] = (rmt_symbol_word_t) {
                .level0 = 1, .duration0 = NEC_PAYLOAD_ONE_DURATION_0,
                .level1 = 0, .duration1 = NEC_PAYLOAD_ONE_DURATION_1
            };
        }
    }
    // Append connecting codes
    items[idx++] = (rmt_symbol_word_t) {
        .level0 = 1, .duration0 = NEC_CONNECTING_CODE1_DURATION_0,
        .level1 = 0, .duration1 = NEC_CONNECTING_CODE1_DURATION_1
    };
    items[idx++] = (rmt_symbol_word_t) {
        .level0 = 1, .duration0 = NEC_CONNECTING_CODE2_DURATION_0,
        .level1 = 0, .duration1 = NEC_CONNECTING_CODE2_DURATION_1
    };
    
    xSemaphoreGive(tx_data_mutex);
    return idx;
}

/* Parse received symbols to binary string */
static void parse_rx_symbols(const rmt_symbol_word_t *symbols, size_t num_symbols, char *output) {
    int idx = 0;
    for (size_t i = 0; i < num_symbols && idx < 64; i++) {
        // Check for payload zero or one based on duration1
        if (abs(symbols[i].duration1 - NEC_PAYLOAD_ZERO_DURATION_1) < EXAMPLE_IR_NEC_DECODE_MARGIN) {
            output[idx++] = '0';
        } else if (abs(symbols[i].duration1 - NEC_PAYLOAD_ONE_DURATION_1) < EXAMPLE_IR_NEC_DECODE_MARGIN) {
            output[idx++] = '1';
        }
    }
    output[idx] = '\0';
    ESP_LOGI(TAG, "Received IR data: %s", output);
}

/* RX callback for IR receiver */
static bool ir_rx_callback(rmt_channel_handle_t channel,
                          const rmt_rx_done_event_data_t *edata,
                          void *user_data) {
    BaseType_t high_task_wakeup = pdFALSE;
    QueueHandle_t receive_queue = (QueueHandle_t)user_data;
    
    // Allocate memory for the received symbols
    rmt_symbol_word_t *symbols_copy = malloc(edata->num_symbols * sizeof(rmt_symbol_word_t));
    if (symbols_copy) {
        memcpy(symbols_copy, edata->received_symbols, edata->num_symbols * sizeof(rmt_symbol_word_t));
        
        // Create IR RX data structure
        ir_rx_data_t *rx_data = malloc(sizeof(ir_rx_data_t));
        if (rx_data) {
            rx_data->symbols = symbols_copy;
            rx_data->num_symbols = edata->num_symbols;
            
            // Send to IR TX task
            xQueueSendFromISR(receive_queue, &rx_data, &high_task_wakeup);
        } else {
            free(symbols_copy);
        }
    }
    
    // Restart reception
    rmt_receive(channel, edata->received_symbols, edata->num_symbols * sizeof(rmt_symbol_word_t), NULL);
    
    return high_task_wakeup == pdTRUE;
}

/* ----------------- WiFi event handlers ----------------- */
static void wifi_event_handler(void* arg, esp_event_base_t event_base,
                               int32_t event_id, void* event_data) {
    if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_START) {
        esp_wifi_connect();
    } else if (event_base == WIFI_EVENT && event_id == WIFI_EVENT_STA_DISCONNECTED) {
        ESP_LOGW(TAG, "WiFi disconnected, reconnecting...");
        esp_wifi_connect();
        xEventGroupClearBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
    } else if (event_base == IP_EVENT && event_id == IP_EVENT_STA_GOT_IP) {
        ip_event_got_ip_t* event = (ip_event_got_ip_t*) event_data;
        ESP_LOGI(TAG, "Got IP: " IPSTR, IP2STR(&event->ip_info.ip));
        xEventGroupSetBits(s_wifi_event_group, WIFI_CONNECTED_BIT);
    }
}

void wifi_init_sta(void) {
    s_wifi_event_group = xEventGroupCreate();

    ESP_ERROR_CHECK(esp_netif_init());
    ESP_ERROR_CHECK(esp_event_loop_create_default());
    esp_netif_create_default_wifi_sta();
    wifi_init_config_t cfg = WIFI_INIT_CONFIG_DEFAULT();
    ESP_ERROR_CHECK(esp_wifi_init(&cfg));

    ESP_ERROR_CHECK(esp_event_handler_instance_register(WIFI_EVENT,
                                                        ESP_EVENT_ANY_ID,
                                                        &wifi_event_handler,
                                                        NULL, NULL));
    ESP_ERROR_CHECK(esp_event_handler_instance_register(IP_EVENT,
                                                        IP_EVENT_STA_GOT_IP,
                                                        &wifi_event_handler,
                                                        NULL, NULL));

    wifi_config_t wifi_config = {
        .sta = {
            .ssid = "",
            .password = "",
            .threshold.authmode = WIFI_AUTH_WPA2_PSK,
            .pmf_cfg = {
                .capable = true,
                .required = false
            }
        },
    };
    strncpy((char*)wifi_config.sta.ssid, WIFI_SSID, sizeof(wifi_config.sta.ssid)-1);
    strncpy((char*)wifi_config.sta.password, WIFI_PASS, sizeof(wifi_config.sta.password)-1);

    ESP_ERROR_CHECK(esp_wifi_set_mode(WIFI_MODE_STA));
    ESP_ERROR_CHECK(esp_wifi_set_config(WIFI_IF_STA, &wifi_config));
    ESP_ERROR_CHECK(esp_wifi_start());

    ESP_LOGI(TAG, "wifi_init_sta finished.");
}

/* ----------------- HTTP client + parsing ----------------- */

/* HTTP event handler */
esp_err_t http_event_handler(esp_http_client_event_t *evt) {
    switch (evt->event_id) {
        case HTTP_EVENT_ON_DATA:
            if (evt->data_len > 0) {
                char *json_str = malloc(evt->data_len + 1);
                if (!json_str) {
                    ESP_LOGE(TAG, "malloc failed");
                    return ESP_FAIL;
                }
                memcpy(json_str, evt->data, evt->data_len);
                json_str[evt->data_len] = '\0';
                ESP_LOGD(TAG, "HTTP DATA: %s", json_str);

                cJSON *root = cJSON_Parse(json_str);
                if (root) {
                    cJSON *commands = cJSON_GetObjectItem(root, "commands");
                    if (cJSON_IsArray(commands)) {
                        int count = cJSON_GetArraySize(commands);
                        ESP_LOGI(TAG, "Found %d commands", count);
                        for (int i = 0; i < count; i++) {
                            cJSON *cmd = cJSON_GetArrayItem(commands, i);
                            if (!cJSON_IsString(cmd)) continue;
                            const char *s = cmd->valuestring;
                            ESP_LOGI(TAG, "cmd: %s", s);

                            // Interpret string into ha_cmd_t
                            ha_cmd_t ocmd = {0};
                            if (strstr(s, "SET_TEMP_") != NULL) {
                                ocmd.type = CMD_SET_TEMP;
                                int val = atoi(s + strlen("SET_TEMP_"));
                                ocmd.int_value = val;
                                xQueueSend(cmd_queue, &ocmd, pdMS_TO_TICKS(10));
                            } else if (strstr(s, "SET_MODE_") != NULL) {
                                ocmd.type = CMD_SET_MODE;
                                const char *mode = s + strlen("SET_MODE_");
                                for (size_t k=0; k<sizeof(ocmd.mode)-1 && mode[k]; k++) {
                                    ocmd.mode[k] = (char)tolower((unsigned char)mode[k]);
                                }
                                xQueueSend(cmd_queue, &ocmd, pdMS_TO_TICKS(10));
                            } else if (strstr(s, "SET_FAN_") != NULL) {
                                ocmd.type = CMD_SET_FAN;
                                int val = atoi(s + strlen("SET_FAN_"));
                                ocmd.int_value = val;
                                xQueueSend(cmd_queue, &ocmd, pdMS_TO_TICKS(10));
                            } else if (strcmp(s, "SWITCH_ON") == 0) {
                                ocmd.type = CMD_SET_POWER;
                                ocmd.int_value = 1;
                                xQueueSend(cmd_queue, &ocmd, pdMS_TO_TICKS(10));
                            } else if (strcmp(s, "SWITCH_OFF") == 0) {
                                ocmd.type = CMD_SET_POWER;
                                ocmd.int_value = 0;
                                xQueueSend(cmd_queue, &ocmd, pdMS_TO_TICKS(10));
                            } else {
                                ESP_LOGW(TAG, "Unknown command: %s", s);
                            }
                        }
                    }
                    cJSON_Delete(root);
                } else {
                    ESP_LOGW(TAG, "JSON parse failed");
                }
                free(json_str);
            }
            break;
        default:
            break;
    }
    return ESP_OK;
}

/* Perform HTTP GET to the HA custom endpoint */
void fetch_commands_from_ha(void) {
    char url[256];
    snprintf(url, sizeof(url),
             "http://%s:%d/api/my_inverter/%s/commands?api_key=%s",
             HA_IP, HA_PORT, HA_ENTRY_ID, HA_API_KEY);

    esp_http_client_config_t config = {
        .url = url,
        .method = HTTP_METHOD_GET,
        .event_handler = http_event_handler,
    };

    esp_http_client_handle_t client = esp_http_client_init(&config);
    esp_http_client_set_header(client, "X-API-Key", HA_API_KEY);

    esp_err_t err = esp_http_client_perform(client);
    if (err == ESP_OK) {
        int status = esp_http_client_get_status_code(client);
        ESP_LOGI(TAG, "HTTP GET status: %d", status);
    } else {
        ESP_LOGE(TAG, "HTTP GET failed: %s", esp_err_to_name(err));
    }

    esp_http_client_cleanup(client);
}

/* HTTP polling task */
static void http_poll_task(void *pvParameter) {
    xEventGroupWaitBits(s_wifi_event_group, WIFI_CONNECTED_BIT, false, true, portMAX_DELAY);
    ESP_LOGI(TAG, "HTTP Poll task: WiFi connected, starting polling...");

    while (1) {
        fetch_commands_from_ha();
        vTaskDelay(pdMS_TO_TICKS(5000));
    }
    vTaskDelete(NULL);
}

/* Command Handler Task - updates tx_data based on HA commands */
static void command_handler_task(void *pvParameter) {
    ESP_LOGI(TAG, "Command Handler task starting");
    
    // Initialize tx_data with default values
    if (xSemaphoreTake(tx_data_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
        strcpy(tx_data, "0100010000000000101001000100000010000000000001010000100000001001");
        xSemaphoreGive(tx_data_mutex);
        ESP_LOGI(TAG, "Initial tx_data set: %s", tx_data);
    }
    
    ha_cmd_t cmd;
    while (1) {
        if (xQueueReceive(cmd_queue, &cmd, portMAX_DELAY) == pdTRUE) {
            ESP_LOGI(TAG, "Processing command type: %d", cmd.type);
            
            // Apply command to update tx_data
            switch (cmd.type) {
                case CMD_SET_TEMP:
                    set_temp(cmd.int_value);
                    break;
                case CMD_SET_MODE:
                    set_mode(cmd.mode);
                    break;
                case CMD_SET_FAN:
                    set_fan(cmd.int_value);
                    break;
                case CMD_SET_POWER:
                    set_power(cmd.int_value);
                    break;
                default:
                    ESP_LOGW(TAG, "Unknown cmd type");
            }
            
            // Log the updated tx_data
            if (xSemaphoreTake(tx_data_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
                ESP_LOGI(TAG, "Updated tx_data: %s", tx_data);
                xSemaphoreGive(tx_data_mutex);
            }
        }
    }
    vTaskDelete(NULL);
}

/* IR TX Task - Listens for IR signals and transmits back using updated tx_data */
static void ir_tx_task(void *pvParameter) {
    ESP_LOGI(TAG, "IR TX/RX task starting");
    
    /* Create RMT RX channel */
    rmt_rx_channel_config_t rx_channel_cfg = {
        .clk_src = RMT_CLK_SRC_DEFAULT,
        .resolution_hz = EXAMPLE_IR_RESOLUTION_HZ,
        .mem_block_symbols = SYMBOL_COUNT,
        .gpio_num = EXAMPLE_IR_RX_GPIO_NUM,
    };
    rmt_channel_handle_t rx_channel = NULL;
    ESP_ERROR_CHECK(rmt_new_rx_channel(&rx_channel_cfg, &rx_channel));
    
    /* Create RX queue for IR signals */
    QueueHandle_t receive_queue = xQueueCreate(1, sizeof(ir_rx_data_t *));
    assert(receive_queue);
    
    rmt_rx_event_callbacks_t cbs = {
        .on_recv_done = ir_rx_callback,
    };
    ESP_ERROR_CHECK(rmt_rx_register_event_callbacks(rx_channel, &cbs, receive_queue));
    
    rmt_receive_config_t receive_config = {
        .signal_range_min_ns = 1250,
        .signal_range_max_ns = 22000000,
    };
    
    ESP_ERROR_CHECK(rmt_enable(rx_channel));
    
    /* Initialize TX channel */
    ESP_LOGI(TAG, "Initializing RMT TX on GPIO %d", EXAMPLE_IR_TX_GPIO_NUM);
    rmt_tx_channel_config_t tx_chan_config = {
        .gpio_num = EXAMPLE_IR_TX_GPIO_NUM,
        .clk_src = RMT_CLK_SRC_DEFAULT,
        .resolution_hz = EXAMPLE_IR_RESOLUTION_HZ,
        .mem_block_symbols = 64,
        .trans_queue_depth = 4,
        .flags.invert_out = 1
    };
    rmt_channel_handle_t tx_chan = NULL;
    ESP_ERROR_CHECK(rmt_new_tx_channel(&tx_chan_config, &tx_chan));
    
    rmt_encoder_handle_t copy_encoder = NULL;
    rmt_copy_encoder_config_t copy_config = {};
    ESP_ERROR_CHECK(rmt_new_copy_encoder(&copy_config, &copy_encoder));
    
    ESP_ERROR_CHECK(rmt_enable(tx_chan));
    
    /* Prepare buffers */
    rmt_symbol_word_t raw_symbols[SYMBOL_COUNT];
    ir_rx_data_t *rx_data_ptr = NULL;
    char binary_data[65] = {0};
    rmt_symbol_word_t items[300];
    rmt_transmit_config_t tx_cfg = { .loop_count = 0 };
    
    /* Start initial reception */
    ESP_ERROR_CHECK(rmt_receive(rx_channel, raw_symbols, sizeof(raw_symbols), &receive_config));
    
    /* Initial 5-second receive-only period */
    TickType_t start_time = xTaskGetTickCount();
    TickType_t rx_period_ticks = pdMS_TO_TICKS(INITIAL_RX_PERIOD_MS);
    bool initial_rx_done = false;
    
    while (!initial_rx_done) {
        if (xQueueReceive(receive_queue, &rx_data_ptr, pdMS_TO_TICKS(100)) == pdPASS) {
            parse_rx_symbols(rx_data_ptr->symbols, rx_data_ptr->num_symbols, binary_data);
            
            // Free the received data
            free(rx_data_ptr->symbols);
            free(rx_data_ptr);
            rx_data_ptr = NULL;
            
            ESP_ERROR_CHECK(rmt_receive(rx_channel, raw_symbols, sizeof(raw_symbols), &receive_config));
        }
        if ((xTaskGetTickCount() - start_time) >= rx_period_ticks) {
            initial_rx_done = true;
        }
    }
    
    ESP_LOGI(TAG, "Initial RX period done. Starting main loop.");
    
    /* Main loop: Receive IR and transmit back with updated tx_data */
    while (1) {
        if (xQueueReceive(receive_queue, &rx_data_ptr, portMAX_DELAY) == pdPASS) {
            // Parse and log received IR data
            parse_rx_symbols(rx_data_ptr->symbols, rx_data_ptr->num_symbols, binary_data);
            
            // Free the received data
            free(rx_data_ptr->symbols);
            free(rx_data_ptr);
            rx_data_ptr = NULL;
            
            // Build and transmit using the current tx_data
            int num_items = build_items(items, sizeof(items)/sizeof(items[0]));
            if (num_items > 0) {
                // Log what we're sending
                if (xSemaphoreTake(tx_data_mutex, pdMS_TO_TICKS(100)) == pdTRUE) {
                    ESP_LOGI(TAG, "Transmitting tx_data: %s", tx_data);
                    xSemaphoreGive(tx_data_mutex);
                }
                
                // Transmit
                esp_err_t err = rmt_transmit(tx_chan, copy_encoder,
                                           items, num_items * sizeof(rmt_symbol_word_t), &tx_cfg);
                if (err != ESP_OK) {
                    ESP_LOGE(TAG, "rmt_transmit failed: %s", esp_err_to_name(err));
                } else {
                    ESP_ERROR_CHECK(rmt_tx_wait_all_done(tx_chan, -1));
                    ESP_LOGI(TAG, "IR transmission complete");
                }
            }
            
            // Restart reception
            ESP_ERROR_CHECK(rmt_receive(rx_channel, raw_symbols, sizeof(raw_symbols), &receive_config));
            
            // Short delay between transmissions
            vTaskDelay(pdMS_TO_TICKS(TX_DELAY_MS));
        }
    }
    vTaskDelete(NULL);
}

/* ----------------- app_main ----------------- */
void app_main(void) {
    ESP_LOGI(TAG, "Starting HA IR Client with RX/TX");
    
    /* Initialize NVS */
    esp_err_t ret = nvs_flash_init();
    if (ret == ESP_ERR_NVS_NO_FREE_PAGES || ret == ESP_ERR_NVS_NEW_VERSION_FOUND) {
        ESP_ERROR_CHECK(nvs_flash_erase());
        ret = nvs_flash_init();
    }
    ESP_ERROR_CHECK(ret);
    
    /* Create mutex for tx_data */
    tx_data_mutex = xSemaphoreCreateMutex();
    configASSERT(tx_data_mutex);
    
    /* Create command queue */
    cmd_queue = xQueueCreate(CMD_QUEUE_LEN, sizeof(ha_cmd_t));
    configASSERT(cmd_queue);
Z
    /* Create IR RX queue */
    ir_rx_queue = xQueueCreate(1, sizeof(ir_rx_data_t *));
    configASSERT(ir_rx_queue);
    
    /* Initialize WiFi */
    wifi_init_sta();
    
    /* Start tasks */
    xTaskCreate(&http_poll_task, "http_poll_task", 8*1024, NULL, 5, NULL);
    xTaskCreate(&command_handler_task, "cmd_handler_task", 8*1024, NULL, 6, NULL);
    xTaskCreate(&ir_tx_task, "ir_tx_task", 8*1024, NULL, 7, NULL);
    
    ESP_LOGI(TAG, "All tasks started. System ready.");
}
