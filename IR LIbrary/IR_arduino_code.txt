#include <IRremoteESP8266.h>
#include <IRsend.h>
#include <ir_Gree.h>  // <-- Gree library

#define IR_LED_PIN 4  // ESP8266 GPIO4 (D2 on NodeMCU)

IRsend irsend(IR_LED_PIN);
IRGreeAC ac(IR_LED_PIN);  // Gree AC object

void setup() {
  irsend.begin();
  Serial.begin(115200);
  while (!Serial);  // Wait for Serial
  Serial.println("Gree AC IR Sender Ready");
}

void loop() {
  ac.on();                          // Power ON
  ac.setMode(kGreeCool);            // Cool mode
  ac.setTemp(18);                   // 22°C = 72°F (Gree uses Celsius internally)
  ac.setFan(kGreeFanMed);           // Medium fan
  ac.setSwingVertical(true, kGreeSwingAuto);  // Swing ON + Auto
  ac.setTurbo(false);
  ac.setLight(true);                // Optional: display ON
  ac.setXFan(false);
  ac.setSleep(false);

  // Send the command 3 times (Gree needs repeats)
  ac.send();
  delay(100);
  ac.send();
  delay(100);
  ac.send();

  Serial.println("Sent: Gree AC -> Cool 72°F (22°C), Fan Med, Swing Auto ON");

  delay(5000);  
}//*/
