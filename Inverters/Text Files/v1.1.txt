#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "esp_system.h"
#include "esp_log.h"
#include "driver/uart.h"
#include "string.h"
#include "driver/gpio.h"
#include <stdlib.h>
#include "freertos/semphr.h"

#define WIFI_SSID "GusetAcess"
#define WIFI_PASS "MRSPAK@ISB"


SemaphoreHandle_t data_mutex;
QueueHandle_t request_queue;
static const int RX_BUF_SIZE = 256;

#define TXD_PIN (GPIO_NUM_4)
#define RXD_PIN (GPIO_NUM_5)
#define NUM_ADDRESSES 28

static const char *TAG = "MODBUS";

void init_uart(void)
{
    const uart_config_t uart_config = {
        .baud_rate = 9600,
        .data_bits = UART_DATA_8_BITS,
        .parity = UART_PARITY_DISABLE,
        .stop_bits = UART_STOP_BITS_1,
        .flow_ctrl = UART_HW_FLOWCTRL_DISABLE,
        .source_clk = UART_SCLK_DEFAULT,
    };

    uart_driver_install(UART_NUM_1, RX_BUF_SIZE * 2, 0, 0, NULL, 0);
    uart_param_config(UART_NUM_1, &uart_config);
    uart_set_pin(UART_NUM_1, TXD_PIN, RXD_PIN, UART_PIN_NO_CHANGE, UART_PIN_NO_CHANGE);
}

uint16_t modbus_crc16(const uint8_t *data, uint16_t length)
{
    uint16_t crc = 0xFFFF;
    for (uint16_t i = 0; i < length; i++) {
        crc ^= data[i];
        for (uint8_t j = 0; j < 8; j++) {
            if (crc & 0x0001) {
                crc >>= 1;
                crc ^= 0xA001;
            } else {
                crc >>= 1;
            }
        }
    }
    return crc;
}


void send_modbus_request(uint16_t start_address)
{
    uint8_t request[8];
    request[0] = 0x01;                      // Slave address
    request[1] = 0x03;                      // Function code (Read Holding Registers)
    request[2] = (start_address >> 8);      // Starting address high byte
    request[3] = (start_address & 0xFF);    // Starting address low byte
    request[4] = 0x00;                      // Number of registers high byte
    request[5] = 0x01;                      // Number of registers low byte

    uint16_t crc = modbus_crc16(request, 6);
    request[6] = crc & 0xFF;                // CRC low byte
    request[7] = (crc >> 8);                // CRC high byte

    uart_write_bytes(UART_NUM_1, (const char *)request, 8);
    ESP_LOGI(TAG, "Sent Request: ");
    ESP_LOG_BUFFER_HEXDUMP(TAG, request, 8, ESP_LOG_INFO);
}


typedef struct 
{
    uint16_t address;
    uint16_t value;
    bool valid;
} modbus_data_t;
modbus_data_t modbus_data[NUM_ADDRESSES];

uint16_t start_addresses[NUM_ADDRESSES] = {0x0256, 0x257, 0x258, 0x26E,0x26F,0x270,0x2AF, 0x2B0, 0x2B1, 0x271, 0x2B2, 0x208, 0x24B, 0x24F, 0x261, 0x262, 0x263, 0x264, 0x284, 0x285, 0x286, 0x28A, 0x28B, 0x28C, 0x290, 0x291, 0x292};
void tx_task(void *arg)
{
    int index = 0;
    while (1) {
        send_modbus_request(start_addresses[index]);

        // Send current index to RX task
        xQueueSend(request_queue, &index, portMAX_DELAY);

        index++;
        if (index >= NUM_ADDRESSES) {
            index = 0;
        }

        vTaskDelay(pdMS_TO_TICKS(1000));
    }
}


void rx_task(void *arg)
{
    uint8_t *data = (uint8_t *)malloc(RX_BUF_SIZE);
    int request_index;

    while (1) {
        if (xQueueReceive(request_queue, &request_index, pdMS_TO_TICKS(1000))) {
            int len = uart_read_bytes(UART_NUM_1, data, RX_BUF_SIZE, pdMS_TO_TICKS(1000));
            if (len > 0) {
                ESP_LOGI(TAG, "Received %d bytes", len);
                ESP_LOG_BUFFER_HEXDUMP(TAG, data, len, ESP_LOG_INFO);

                // Validate basic response structure and CRC
                if (len >= 7 && data[0] == 0x01 && data[1] == 0x03 && data[2] == 0x02) {
                    uint16_t crc_calc = modbus_crc16(data, len - 2);
                    uint16_t crc_recv = (data[len - 1] << 8) | data[len - 2];

                    if (crc_calc == crc_recv) {
                        uint16_t value = (data[3] << 8) | data[4];

                        xSemaphoreTake(data_mutex, portMAX_DELAY);
                        modbus_data[request_index].value = value;
                        modbus_data[request_index].valid = true;
                        xSemaphoreGive(data_mutex);
                    } else {
                        ESP_LOGW(TAG, "Invalid CRC: Calculated=0x%04X Received=0x%04X", crc_calc, crc_recv);
                    }
                }
            }
        }
    }

    free(data);
}

void print_task(void *arg)
{
    while (1) {
        vTaskDelay(pdMS_TO_TICKS(3000));  
        xSemaphoreTake(data_mutex, portMAX_DELAY);
        printf("========= MODBUS DATA =========\n");
        for (int i = 0; i < NUM_ADDRESSES; i++) {
            if (modbus_data[i].valid) {
                printf("Address 0x%04X: Value = %u\n", modbus_data[i].address, modbus_data[i].value);
            } else {
                printf("Address 0x%04X: Value = N/A\n", modbus_data[i].address);
            }
        }
        printf("================================\n");
        xSemaphoreGive(data_mutex);
    }
}



void app_main(void)
{
    init_uart();

    data_mutex = xSemaphoreCreateMutex();
    request_queue = xQueueCreate(10, sizeof(int));

    // Initialize storage
    for (int i = 0; i < NUM_ADDRESSES; i++) {
        modbus_data[i].address = start_addresses[i];
        modbus_data[i].valid = false;
    }

    xTaskCreate(rx_task, "modbus_rx_task", 4096, NULL, 10, NULL);
    xTaskCreate(tx_task, "modbus_tx_task", 4096, NULL, 9, NULL);
    xTaskCreate(print_task, "print_task", 4096, NULL, 8, NULL);
}
